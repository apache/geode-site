417  inline void reset() {
418  if (m_haveBigBuffer) {
419  // create smaller buffer
420  m_bytes.reset(
421  static_cast<uint8_t*>(std::malloc(m_lowWaterMark * sizeof(uint8_t))));
422  if (m_bytes == nullptr) {
423  throw OutOfMemoryException("Out of Memory while resizing buffer");
424  }
425  m_size = m_lowWaterMark;
426  // reset the flag
427  m_haveBigBuffer = false;
428  // release the lock
429  releaseLock();
430  }
431  m_buf = m_bytes.get();
432  }
433 
434  // make sure there is room left for the requested size item.
435  inline void ensureCapacity(size_t size) {
436  size_t offset = m_buf - m_bytes.get();
437  if ((m_size - offset) < size) {
438  size_t newSize = m_size * 2 + (8192 * (size / 8192));
439  if (newSize >= m_highWaterMark && !m_haveBigBuffer) {
440  // acquire the lock
441  acquireLock();
442  // set flag
443  m_haveBigBuffer = true;
444  }
445  m_size = newSize;
446 
447  auto bytes = m_bytes.release();
448  auto tmp =
449  static_cast<uint8_t*>(std::realloc(bytes, m_size * sizeof(uint8_t)));
450  if (tmp == nullptr) {
451  throw OutOfMemoryException("Out of Memory while resizing buffer");
452  }
453  m_bytes.reset(tmp);
454  m_buf = m_bytes.get() + offset;
455  }
456  }

559  inline void writeJavaModifiedUtf8(const char16_t* data, size_t len) {
560  if (0 == len) {
561  writeInt(static_cast<uint16_t>(0));
562  } else {
563  auto encodedLen = static_cast<uint16_t>(
564  std::min<size_t>(getJavaModifiedUtf8EncodedLength(data, len),
565  std::numeric_limits<uint16_t>::max()));
566  writeInt(encodedLen);
567  ensureCapacity(encodedLen);
568  const auto end = m_buf + encodedLen;
569  while (m_buf < end) {
570  encodeJavaModifiedUtf8(*data++);
571  }
572  if (m_buf > end) m_buf = end;
573  }
574  }

648  inline static void getEncodedLength(const char val, int32_t& encodedLen) {
649  if ((val == 0) || (val & 0x80)) {

652  } else {
653  // one byte.
654  encodedLen++;
655  }
656  }

658  inline static void getEncodedLength(const wchar_t val, int32_t& encodedLen) {
659  if (val == 0) {
660  encodedLen += 2;
661  } else if (val < 0x80) // ASCII character
662  {
663  encodedLen++;
664  } else if (val < 0x800) {
665  encodedLen += 2;
666  } else {
667  encodedLen += 3;
668  }

671  inline void encodeChar(const char value) {
672  uint8_t tmp = static_cast<uint8_t>(value);
673  if ((tmp == 0) || (tmp & 0x80)) {
674  // two byte.
675  *(m_buf++) = static_cast<uint8_t>(0xc0 | ((tmp & 0xc0) >> 6));
676  *(m_buf++) = static_cast<uint8_t>(0x80 | (tmp & 0x3f));
677  } else {
678  // one byte.
679  *(m_buf++) = tmp;
680  }
681  }
682 
683  // this will lose the character set encoding.
684  inline void encodeChar(const wchar_t value) {
685  uint16_t c = static_cast<uint16_t>(value);
686  if (c == 0) {
687  *(m_buf++) = 0xc0;
688  *(m_buf++) = 0x80;
689  } else if (c < 0x80) { // ASCII character
690  *(m_buf++) = static_cast<uint8_t>(c);
691  } else if (c < 0x800) {
692  *(m_buf++) = static_cast<uint8_t>(0xC0 | c >> 6);
693  *(m_buf++) = static_cast<uint8_t>(0x80 | (c & 0x3F));
694  } else {
695  *(m_buf++) = static_cast<uint8_t>(0xE0 | c >> 12);
696  *(m_buf++) = static_cast<uint8_t>(0x80 | ((c >> 6) & 0x3F));
697  *(m_buf++) = static_cast<uint8_t>(0x80 | (c & 0x3F));
698  }
699  }

702  if (c == 0) {

704  *(m_buf++) = 0xc0;
